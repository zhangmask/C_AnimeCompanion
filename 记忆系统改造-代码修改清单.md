@Query("""
    UPDATE memories SET strength = CASE
        WHEN :idleDays >= 4 THEN ROUND(strength * 0.90, 4)
        WHEN :idleDays = 3  THEN ROUND(strength * 0.90, 4)
        WHEN :idleDays = 2  THEN ROUND(strength * 0.80, 4)

### 14. `MemoryDecayManager.kt` — 新增衰减调度器

**文件路径：** `app/src/main/java/com/companion/chat/data/memory/MemoryDecayManager.kt`（从方案 v1 已有，v2 微调衰减参数）

```kotlin
class MemoryDecayManager(
    private val memoryDao: MemoryDao,
    private val nowProvider: () -> Long = { System.currentTimeMillis() }
) {
    suspend fun applyDailyDecay(): Int {
        val now = nowProvider()
        // 遍历所有 strength > 0.05 的记忆，按 idleDays 衰减
        val allMemories = memoryDao.getActiveMemories(0.05)
        var count = 0
        for (memory in allMemories) {
            val idleDays = (now - memory.lastAccessedAt) / DAY_MILLIS
            if (idleDays >= 1) {
                memoryDao.applyDecayByAge(memory.id, idleDays.toInt(), now)
                count++
            }
        }
        return count
    }

    suspend fun strengthenMemory(memoryId: Long, delta: Double = 0.15) {
        memoryDao.strengthen(memoryId, delta, nowProvider())
    }

    suspend fun cleanupWeakMemories(threshold: Double = 0.05): Int {
        return memoryDao.deleteByStrengthBelow(threshold)
    }

    companion object {
        private const val DAY_MILLIS = 24L * 60 * 60 * 1000
    }
}
```

**v2 参数调整**：衰减曲线由 `×0.95/天` 改为分段式（第1天×0.70，第2天×0.80，第3天×0.90，第4天起 ×0.90/天）。

suspend fun buildLayered(
    memories: List<Memory>,
    tokenBudget: Int = 1200,
    lang: AppLanguage = LocalLanguage.current
): String {
    // 按 final_score 降序
    val sorted = memories.sortedByDescending { it.strength }
    
    // 先塞 L0 摘要（低成本快速扫描）
    val l0Section = buildSummarySection(sorted, lang)
    
    // 再塞 L1 概览（按 token 预算裁剪）
    var usedTokens = estimateTokens(l0Section)
    val l1Items = mutableListOf<Memory>()
    for (memory in sorted) {
        val text = memory.l1Overview ?: memory.content
        val tokens = estimateTokens(text)
        if (usedTokens + tokens > tokenBudget) break
        l1Items.add(memory)
        usedTokens += tokens
    }
    
    return buildString {
        appendLine(l0Section)
        if (l1Items.isNotEmpty()) {
            appendLine(buildSection(
                title = Strings.get(lang, StringsKey.memory_retrieved_title),
                memories = l1Items
            ))
        }
    }
}
